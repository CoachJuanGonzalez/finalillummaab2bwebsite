function determineCustomerTier(units, readiness) {
    const unitCount = parseInt(units) || 0;
    
    // Debug log
    console.log('Determining tier - unitCount:', unitCount, 'readiness:', readiness);
    
    // IMPORTANT: Check readiness FIRST
    if (readiness === 'researching' || readiness === 'planning-long' || unitCount === 0) {
        return 'tier_0_explorer';
    }
    
    // Then check unit counts
    if (unitCount >= 1 && unitCount <= 49) return 'tier_1_starter';
    if (unitCount >= 50 && unitCount <= 149) return 'tier_2_pioneer';
    if (unitCount >= 150 && unitCount <= 299) return 'tier_3_preferred';
    if (unitCount >= 300) return 'tier_4_elite';
    
    return 'tier_0_explorer'; // fallback
}
